    ```
    $ tar zxvf glibc-2.24.tar.gz
    $ cd glibc-2.24
    $ mkdir build
    $ cd build
    $ ../configure --prefix=/opt/glibc-2.24
    $ make
    $ sudo make install
    $ export LD_LIBRARY_PATH=/opt/glibc-2.24/lib
    ```
